--Copy Paste all the queries in order to create the database for the project

create database bankModel;
use bankModel;

CREATE TABLE accTable (
accoNo VARCHAR(100) PRIMARY KEY NOT NULL,
accType VARCHAR(100),
balance double,
employmentSta VARCHAR(100)
);


CREATE TABLE custInfoTable (
    accoNo VARCHAR(100) UNIQUE,
    name VARCHAR(100),
    fatherName VARCHAR(100),
    dob VARCHAR(100),
    gender CHAR(1),
    email VARCHAR(100),
    address VARCHAR(100),
    mailingAddress VARCHAR(100),
    province VARCHAR(100),
    city VARCHAR(100),
    postalCode INT,
    cnic VARCHAR(100),
    religon VARCHAR(100),
    martialSta VARCHAR(100),
    FOREIGN KEY (accoNo) references accTable (accoNo)
);



CREATE TABLE SecuredInfoTable(
accoNo VARCHAR(100) PRIMARY KEY NOT NULL,
mpin varchar(200),
pin int,
FOREIGN KEY (accoNo) references accTable (accoNo)
);
